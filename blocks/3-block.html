CREATE OR REPLACE FUNCTION
  valid_deptid (depID Departments.Department_ID%TYPE)
RETURN BOOLEAN IS 
BEGIN
  FOR i IN (
      SELECT 'Exists' FROM dual
      WHERE EXISTS (
        SELECT Department_ID
        FROM Departments
        WHERE Department_ID = depID
      )
    )
  LOOP
    RETURN TRUE;
  END LOOP;
  RETURN FALSE;
END valid_deptid;
